2. Dane

Pn = 54000;  %moc znamionowa [W]
Un = 440;   %napięcie znamionowe [V]
In = 130.7;    %prąd znamionowy [A]
Nn = 1450;  %prędkość znamionowa [obr/min]
R = 0.254;  %rezystancja twornika [Ohm]
L = 1.63*10^-3;   %indukcujnosc twornika [H]
Js = 0.97*15;  %moment bezwladnosci [kg*m^2]

3. Przełącznia układu

regpred = 1;        %1 - regulator PI, 0 - reg P
przelacz3 = 0;      %1 - załącza moment obciążenia
delomega = 0;       %1 - deltaomega 2%, 0 - 5%

4. Sprawdzenie warunków

fn = Nn/60;     %częstotliwość znamionowa
omegan = 2*pi*fn;   %pulsacja znamionowa
lambdan = 2;
p = 50;
T = L/R;    %elektromagnetyczna stała czasowa
J = 6*Js;   %zastępszy moment bezwładności
psien = (Un-R*In)/omegan;   %znamionowy strumień
B = J*R/psien^2;    %elektromechaniczna stała czasowa

5. Obliczenia

tau0 = 3.3e-3;  %[s] stała czasowa przekształtnika tyrystorowego
Kp = 1.5*Un/10;    %wzmocnienie wzmacniacza mocy
Y = 7.8/(2.5*In);     %zmienione z 10 na 7.8, bo nie spełniało warunku z Uz0
Kt = 10/(1.2*omegan);   %wzmocnienie
Mn = psien*In;  %znamionowy moment obciążenia
Top = 0;  %opóźnienie załączanie znamionowego momentu obciążenia
if (delomega == 0)
    deltaomega = 0.05*omegan;   %statyzm regulacji
else
    deltaomega = 0.02*omegan;
end;
beta = lambdan/p;   %stała czasowa przebiegu prądu twornika
T1 = 0.5*B*(1-sqrt(1-4*T/B));
B1 = B-T1;
kz = (B1-beta)/(Y*B1);  %zastępczy współczynnik wzmocnienia
if (B > 4*T)
    m = T1;                         %parametry regulatora
    V = beta*Y*Kp*B/((B1-beta)*R);  %
else
    m = sqrt(B*T);
    V = Kp*Y/R*B*beta/(sqrt(B*T)-beta);
end
Uz0 = lambdan*In*Y*B1/(B1-beta);    %ograniczenie Uz0 < 10V

TR = 4*beta;  %parametr wg kryterium symetrycznego
if (regpred == 0)
    Kw = Mn/(psien*kz*Kt*deltaomega);    %regulator P
else
    Kw = J/(2*Kt*kz*beta*psien);    %regulator PI
end;
time = linspace(0,1,2);
Imax = In*lambdan+0*time;

8. Przekształtnik tyrystrowy

KR=m/V;
tau0=3.3*10^(-3);
reg_pred=tf([KR*TR KR], [TR 0]);        %regulacja predkosci
reg_mom=tf([m 1], [V 0]);               %regulacja momentu
przeksztyr=tf([Kp], [tau0 1]);          %przeksztaltnik tyrystorowy
silnik=tf([B 0], [R*B1*T1 R*(B1+T1) R]);    %moment silnika
bezwl=tf([1],[J 0]);                    %bezwladnosc

sys=series(reg_mom,przeksztyr);
sys=series(sys,silnik);

sys=feedback(sys,Y);

sys=series(psien,sys);
sys=series(reg_pred,sys);
sys=series(sys,bezwl);

figure
bode(sys)
grid on;

P = nyquistoptions;
P.ShowFullContour = 'off';

figure
nyquist(sys, P)
grid on

figure
margin(sys)

% op=Pm*(pi/180)/Wpm;

Gp = tf(Kp,[tau0 1]);
Gi = tf(m+1, V);
Gs = tf(B/R,[B1 1]);
G1 = Kt*tf([TR 1],[TR 0]);
Gj = tf(1,[J 1]);

Gz = Gi*Gp*Gs/((1+Gi*Gp*Gs)*Y);
G0 = G1*Gz*psien*Gj;

figure()
margin(G0);

%Charakterystyki
figure()   %dla Gz
subplot(211)
title('Charakterystyki dla Gz');
hold on
bode(Gz);
grid on
P = nyquistoptions;
P.ShowFullContour = 'off';
subplot(212)
nyquist(Gz, P);
figure()   %dla G0
subplot(211)
title('Charakterystyki dla G0');
hold on
bode(G0);
grid on
subplot(212)
nyquist(G0, P);

9. Dyskretny regulator prędkości/prądu

N = 10; %liczba próbek
Tp = beta/N;    %krok próbkowania
K1 = Kw;    %wzmocnienia regulatora prędkości
K2 = Kw*(1-(Tp/TR));

K1I = m/V;  %wzmocnienia regulatora prądu
K2I = (Tp-m)/V;

deltaI = psien*V*Mn/(psien^2*V+J*Kp*Y);
